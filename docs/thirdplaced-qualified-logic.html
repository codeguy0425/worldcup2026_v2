qualified — threat counting algorithm

condition qualified = threats < 8

A threat is another third-placed team that could mathematically finish ahead of you.

// For each opponent 'o' against target 'e': const maxPts = o.pts + (o.thirdLocked ? 0 : 3) if (maxPts > e.pts) → threat if (maxPts < e.pts) → safe (cannot reach) // maxPts === e.pts — tied on points: if (o.thirdLocked) { return isAheadOnTiebreakers(o, e) ← fix: actual tiebreaker check } return true ← not locked: could improve GD/GF, count as threat

tiebreaker isAheadOnTiebreakers(a, b)

function isAheadOnTiebreakers(a, b) { // Pts → GD → GF → Fair Play → drawing of lots (teamId) const cmp = b.pts - a.pts || b.gd - a.gd || b.gf - a.gf if (cmp !== 0) return cmp < 0 const fpA = fairPlayScores[a.teamId] ?? 0 const fpB = fairPlayScores[b.teamId] ?? 0 if (fpA !== fpB) return fpA > fpB return a.teamId < b.teamId }
eliminated — definitive check algorithm

condition eliminated = definitivelyAhead >= 8

A team is definitively ahead if their current points (worst case — lose all remaining) already exceed your maximum possible points (best case — win all remaining).

const myMax = e.pts + (e.thirdLocked ? 0 : 3) const definitivelyAhead = entries.filter(o => { if (o === e) return false return o.pts > myMax // Their floor > your ceiling }).length e.eliminated = definitivelyAhead >= 8
